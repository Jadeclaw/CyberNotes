Get-ADDomainController -credential <username>@<domain>.<domain> # Afficher les informations principales de l'AD (nécessite authentification)

Get-ADDomainController -filter {isGlobalcatalog -eq $true} # Afficher les DC du catalogue global

Get-ADDomain # Affiche toutes les informations de l'AD

Get-ADforest # Afficher tout les domaines dans sa forêt

Get-ADComputer -filter * # Affiche tout les membres du domaine
